cmake_minimum_required(VERSION 3.20)
project(mpf-generated-code-test LANGUAGES CXX)

if(NOT DEFINED GENERATED_SOURCE)
  message(FATAL_ERROR "GENERATED_SOURCE is required")
endif()

if(GENERATED_COMPILE_ONLY)
  add_library(mpf-generated-program OBJECT "${GENERATED_SOURCE}")
else()
  add_executable(mpf-generated-program "${GENERATED_SOURCE}")
endif()
target_compile_features(mpf-generated-program PRIVATE cxx_std_17)
set_target_properties(mpf-generated-program PROPERTIES CXX_EXTENSIONS OFF)
if(MSVC)
  target_compile_options(mpf-generated-program PRIVATE /W4 /WX /permissive-)
else()
  target_compile_options(mpf-generated-program PRIVATE -Wall -Wextra -Wpedantic -Werror)
endif()
